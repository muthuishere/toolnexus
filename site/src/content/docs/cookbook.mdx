## 1. Zero to agent

The three-liner: a toolkit with the ten built-in tools, a client pointed at any
OpenAI- or Anthropic-style endpoint, and one `run`.

<Tabs syncKey="lang">
<TabItem label="JavaScript">
```js
import { createToolkit, createClient } from "toolnexus"

const tk = await createToolkit()                       // 10 built-ins, on by default
const agent = createClient({
  baseUrl: "https://openrouter.ai/api/v1", style: "openai",
  model: "openai/gpt-4o-mini", apiKey: process.env.OPENROUTER_API_KEY,
})
const res = await agent.run("List the files here, then read the largest one.", tk)
console.log(res.text)
```
</TabItem>
<TabItem label="Python">
```python
from toolnexus import create_toolkit, create_client

tk = await create_toolkit()                            # 10 built-ins, on by default
agent = create_client(
    base_url="https://openrouter.ai/api/v1", style="openai",
    model="openai/gpt-4o-mini",
)
res = await agent.run("List the files here, then read the largest one.", tk)
print(res.text)
```
</TabItem>
<TabItem label="Go">
```go

    Model: "openai/gpt-4o-mini", APIKey: os.Getenv("OPENROUTER_API_KEY"),
})

fmt.Println(res.Text)
```
</TabItem>

## 2. MCP servers + agent skills

The two headline dynamic sources. Point the toolkit at an `mcp.json` (local stdio
**and** remote streamable-HTTP servers) and a folder of `**/SKILL.md` skills — both
join the same tool set. A failing MCP server is isolated; it never breaks the build.

<Tabs syncKey="lang">
<TabItem label="JavaScript">
```js
const tk = await createToolkit({
  mcpConfig: "./mcp.json",        // path, raw JSON string, or a parsed object
  skillsDir: "./skills",          // one or more skill roots
})
console.log(tk.tools().map((t) => `${t.name} (${t.source})`))
```
</TabItem>
<TabItem label="Python">
```python
tk = await create_toolkit(
    mcp_config="./mcp.json",       # path, raw JSON string, or a parsed dict
    skills_dir="./skills",         # one or more skill roots
)
print([f"{t.name} ({t.source})" for t in tk.tools()])
```
</TabItem>
<TabItem label="Go">
```go
tk, _ := toolnexus.CreateToolkit(ctx, toolnexus.Options{
    McpConfig: "./mcp.json",                      // path, raw JSON, or parsed map
    SkillsDir: []string{"./skills"},              // one or more skill roots
})
```
</TabItem>
<TabItem label="Java">
```java
Toolkit tk = Toolkit.create(new Toolkit.Options()
        .mcpConfig("mcp.json")        // path, raw JSON string, or a parsed Map
        .skillsDir("./skills"));      // one or more skill roots
```
</TabItem>
<TabItem label="C#">
```csharp
await using var tk = await Toolkit.CreateAsync(new Toolkit.Options()
    .WithMcpConfig("./mcp.json")     // path, raw JSON string, or a parsed dict
    .WithSkillsDir("./skills"));     // one or more skill roots
```
</TabItem>
<TabItem label="Elixir">
```elixir

  mcp_config: "./mcp.json",          # path, raw JSON string, or a parsed map
  skills_dir: "./skills"             # one or more skill roots
)
```
</TabItem>

## 3. Add one of your own functions as a tool

A native tool wraps a plain function: return a string (→ output) or throw (→ error
result). It joins the toolkit like any other tool.

<Tabs syncKey="lang">
<TabItem label="JavaScript">
```js
import { defineTool } from "toolnexus"

tk.register(defineTool({
  name: "add", description: "Add two numbers and return the sum.",
  inputSchema: { type: "object", properties: { a: { type: "number" }, b: { type: "number" } }, required: ["a", "b"] },
  run: ({ a, b }) => `${Number(a) + Number(b)}`,
}))
```
</TabItem>
<TabItem label="Python">
```python
from toolnexus import define_tool

def add(a: float, b: float) -> str:
    """Add two numbers and return the sum."""
    return str(a + b)

tk.register(define_tool(add, name="add"))
```
</TabItem>
<TabItem label="Go">
```go
type addArgs struct { A float64 `json:"a"`; B float64 `json:"b"` }

tk.Register(toolnexus.NativeToolReflect("add",
    "Add two numbers and return the sum.",
    func(_ context.Context, in addArgs) (string, error) {
        return strconv.FormatFloat(in.A+in.B, 'f', -1, 64), nil
    }))
```
</TabItem>
<TabItem label="Java">
```java
import io.github.muthuishere.toolnexus.NativeTool;
import java.util.Map;

Map<String, Object> schema = Map.of("type", "object",
        "properties", Map.of("a", Map.of("type", "number"), "b", Map.of("type", "number")),
        "required", java.util.List.of("a", "b"));

tk.register(NativeTool.of("add", "Add two numbers and return the sum.", schema,
        args -> String.valueOf(((Number) args.get("a")).doubleValue()
                             + ((Number) args.get("b")).doubleValue())));
```
</TabItem>
<TabItem label="C#">
```csharp
tk.Register(NativeTool.Of("add", "Add two numbers and return the sum.",
    new Dictionary<string, object?> {
        ["type"] = "object",
        ["properties"] = new Dictionary<string, object?> {
            ["a"] = new Dictionary<string, object?> { ["type"] = "number" },
            ["b"] = new Dictionary<string, object?> { ["type"] = "number" } },
        ["required"] = new List<object?> { "a", "b" },
    },
    args => (Convert.ToDouble(args["a"]) + Convert.ToDouble(args["b"])).ToString()));
```
</TabItem>
<TabItem label="Elixir">
```elixir
add = Toolnexus.define_tool(
  name: "add", description: "Add two numbers and return the sum.",
  input_schema: %{"type" => "object",
    "properties" => %{"a" => %{"type" => "number"}, "b" => %{"type" => "number"}},
    "required" => ["a", "b"]},
  execute: fn %{"a" => a, "b" => b}, _ctx -> to_string(a + b) end
)
tk = Toolnexus.Toolkit.register(tk, [add])
```
</TabItem>
</Tabs>

## 4. Turn a REST endpoint into a tool

`{placeholder}` in the URL is filled from the args; `${ENV}` in a header expands
from the environment at call time and is never logged. Non-2xx becomes an
`HTTP <status>: <body>` error result.

<Tabs syncKey="lang">
<TabItem label="JavaScript">
```js
import { httpTool } from "toolnexus"

tk.register(httpTool({
  name: "get_post", description: "Fetch a placeholder blog post by id.",
  method: "GET", url: "https://jsonplaceholder.typicode.com/posts/{id}",
  inputSchema: { type: "object", properties: { id: { type: "number" } }, required: ["id"] },
}))
```
</TabItem>
<TabItem label="Python">
```python
from toolnexus import http_tool

tk.register(http_tool(
    name="get_post", description="Fetch a placeholder blog post by id.",
    method="GET", url="https://jsonplaceholder.typicode.com/posts/{id}",
    input_schema={"type": "object", "properties": {"id": {"type": "number"}}, "required": ["id"]},
))
```
</TabItem>
<TabItem label="Go">
```go
tk.Register(toolnexus.HTTPTool(toolnexus.HTTPToolOptions{
    Name: "get_post", Description: "Fetch a placeholder blog post by id.",
    Method: "GET", URL: "https://jsonplaceholder.typicode.com/posts/{id}",
    InputSchema: toolnexus.JSONSchema{
        "type": "object",
        "properties": map[string]any{"id": map[string]any{"type": "number"}},
        "required": []string{"id"}},
}))
```
</TabItem>
<TabItem label="Java">
```java
import io.github.muthuishere.toolnexus.HttpTool;

HttpTool.Options o = new HttpTool.Options();
o.name = "get_post";
o.description = "Fetch a placeholder blog post by id.";
o.method = "GET";
o.url = "https://jsonplaceholder.typicode.com/posts/{id}";
o.inputSchema = Map.of("type", "object",
        "properties", Map.of("id", Map.of("type", "number")),
        "required", java.util.List.of("id"));
tk.register(HttpTool.of(o));
```
</TabItem>
<TabItem label="C#">
```csharp
tk.Register(HttpTool.Of(new HttpTool.Options {
    Name = "get_post", Description = "Fetch a placeholder blog post by id.",
    Method = "GET", Url = "https://jsonplaceholder.typicode.com/posts/{id}",
    InputSchema = new Dictionary<string, object?> {
        ["type"] = "object",
        ["properties"] = new Dictionary<string, object?> { ["id"] = new Dictionary<string, object?> { ["type"] = "number" } },
        ["required"] = new List<object?> { "id" },
    },
}));
```
</TabItem>
<TabItem label="Elixir">
```elixir
get_post = Toolnexus.Http.tool(
  name: "get_post", description: "Fetch a placeholder blog post by id.",
  method: "GET", url: "https://jsonplaceholder.typicode.com/posts/{id}",
  input_schema: %{"type" => "object",
    "properties" => %{"id" => %{"type" => "integer"}}, "required" => ["id"]}
)
tk = Toolnexus.Toolkit.register(tk, [get_post])
```
</TabItem>
</Tabs>

## 5. Multi-turn memory

Keep a conversation across turns. JS / Python / Go use a `conversation` object whose
`send` retains history; Java / C# / Elixir use `ask` with a stable id whose transcript
lives in the client's conversation store. Both remember turn 1 at turn 2.

<Tabs syncKey="lang">
<TabItem label="JavaScript">
```js
const convo = agent.conversation({ toolkit: tk })
await convo.send("My name is Muthu. Remember it.")
const r = await convo.send("What's my name?")
console.log(r.text)   // recalls "Muthu"
```
</TabItem>
<TabItem label="Python">
```python
convo = agent.conversation(tk)
await convo.send("My name is Muthu. Remember it.")
r = await convo.send("What's my name?")
print(r.text)   # recalls "Muthu"
```
</TabItem>
<TabItem label="Go">
```go
convo := agent.Conversation(tk)
convo.Send(ctx, "My name is Muthu. Remember it.")
r, _ := convo.Send(ctx, "What's my name?")
fmt.Println(r.Text)   // recalls "Muthu"
```
</TabItem>
<TabItem label="Java">
```java
agent.ask("My name is Muthu. Remember it.", tk, "user-42");
LlmClient.RunResult r = agent.ask("What's my name?", tk, "user-42");
System.out.println(r.text);   // recalls "Muthu"
```
</TabItem>
<TabItem label="C#">
```csharp
await agent.AskAsync("My name is Muthu. Remember it.", tk, "user-42");
var r = await agent.AskAsync("What's my name?", tk, "user-42");
Console.WriteLine(r.Text);   // recalls "Muthu"
```
</TabItem>
<TabItem label="Elixir">
```elixir
Toolnexus.Client.ask(client, "My name is Muthu. Remember it.", tk, "user-42")
r = Toolnexus.Client.ask(client, "What's my name?", tk, "user-42")
IO.puts(r.text)   # recalls "Muthu"
```
</TabItem>
